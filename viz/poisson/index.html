library(mapproj)
library(ggthemes)
library(maps)
library(animation)
library(dplyr)
library(purrr)
library(readr)
library(tidyr)
library(tibble)
library(ggplot2)
library(tidyverse)
library(gmailr)
for (n in 1:25) {
    means = rep(NA, 1e+05)
    for (i in 1:1e+05) {
        means[i] = mean(rpois(n, 2))
    }
    print(ggplot() + geom_histogram(aes(x = means), binwidth = 1/n, 
        color = "white", fill = "darkred") + ggtitle(paste0("Poisson(2) Sampling Distribution (n=", 
        as.character(n), ")")) + xlim(-1, 6))
}
